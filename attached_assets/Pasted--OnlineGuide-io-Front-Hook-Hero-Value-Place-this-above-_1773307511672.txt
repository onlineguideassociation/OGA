🗺 Google Maps Integration

Add Google Maps layer under your tourism intelligence nodes.

Example script for your project:

<div id="map"></div>

<script>
function initMap() {

const cambodia = { lat: 12.5657, lng: 104.9910 };

const map = new google.maps.Map(document.getElementById("map"), {
zoom: 7,
center: cambodia,
styles: [
{ elementType: "geometry", stylers: [{ color: "#0b1f3a" }] },
{ elementType: "labels.text.fill", stylers: [{ color: "#ffffff" }] }
]
});

const angkor = { lat: 13.4125, lng: 103.8667 };

new google.maps.Marker({
position: angkor,
map: map,
title: "Angkor Wat"
});

}
</script>

<script async
src="https://maps.googleapis.com/maps/api/js?key=YOUR_API_KEY&callback=initMap">
</script>
